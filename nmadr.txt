
model Reviews {
  id        String   @id @default(uuid())
  name      String
  phone     String
  email     String
  message   String
  createdAt DateTime @default(now())
}

model TopCategory {
  id         String     @id @default(uuid())
  name       String
  createdAt  DateTime   @default(now())
  updateAt   DateTime   @updatedAt
  categories Category[]
}

model Category {
  id            String       @id @default(uuid())
  name          String
  createdAt     DateTime     @default(now())
  updateAt      DateTime     @updatedAt
  topCategory   TopCategory? @relation(fields: [topCategoryId], references: [id])
  topCategoryId String?
  products      Product[]
}

model Product {
  id          String    @id @default(uuid())
  name        String
  description String
  feature     String
  price       String
  brand       String
  image       String[]
  createdAt   DateTime  @default(now())
  updateAt    DateTime  @updatedAt
  category    Category? @relation(fields: [categoryId], references: [id])
  categoryId  String?
}
